--- PAGE 10 ---
Intial Definition, Extraction, Transaformation, and 
Loading for Fact table 
Note: in this implementation surrogate key from DimStore is not used. 
Pratice exercie: change this code to use surrogate key from DimStore instead of using store_id. 
Create table sakila_dwh.FactRental AS 
SELECT date(r.rental_date) as datekey, i.store_id, dc.categorykey, count(1) as num_rental 
FROM sakila.rental r  
 JOIN sakila.inventory i ON r.inventory_id = i.inventory_id  
 JOIN sakila.film f ON i.film_id = f.film_id  
     JOIN sakila.film_category fc ON f.film_id = fc.film_id     
 JOIN sakila.category c ON fc.category_id=c.category_id 
     JOIN sakila_dwh.dimcategory dc on c.category_id = dc.category_id 
WHERE r.rental_date<'2005-08-01' 
GROUP BY datekey, store_id, categorykey 
ORDER BY datekey

--- PAGE 13 ---
Daily ETL 
 
 
 
 
 
 
INSERT INTO sakila_dwh.FactRental  
SELECT date(r.rental_date) as datekey, i.store_id, dc.categorykey, count(1) as num_rental 
FROM sakila.rental r   
 JOIN sakila.inventory i ON r.inventory_id = i.inventory_id   
 JOIN sakila.film f ON i.film_id = f.film_id       
 JOIN sakila.film_category fc ON f.film_id = fc.film_id      
 JOIN sakila.category c ON fc.category_id=c.category_id      
 JOIN sakila_dwh.dimcategory dc on c.category_id = dc.category_id 
WHERE r.rental_date>='2005-08-01' AND r.rental_date<'2005-08-02' 
GROUP BY datekey, store_id, categorykey 
ORDER BY datekey
